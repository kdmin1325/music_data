server {
    listen 443 ssl;
    server_name findipp.ddns.net;

    ssl_certificate /etc/letsencrypt/live/findipp.ddns.net-0001/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/findipp.ddns.net-0001/privkey.pem;
    include /etc/letsencrypt/options-ssl-nginx.conf;
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem;

    location / {
        proxy_pass http://findipp.ddns.net:5000; #for demo purposes
        proxy_set_header    Host                $http_host;
        proxy_set_header    X-Real-IP           $remote_addr;
        proxy_set_header    X-Forwarded-For     $proxy_add_x_forwarded_for;
    }

}
